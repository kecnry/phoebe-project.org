PHOEBE_ast

typedef struct {
	enum {
		PHOEBE_AST_INDEX,
		PHOEBE_AST_NUMVAL,
		PHOEBE_AST_STRING,
		PHOEBE_AST_PARAMETER,
		PHOEBE_AST_NODE
	} type;
	union {
		int    idx;
		double numval;
		char  *str;
		struct {
			PHOEBE_node_type        type;
			struct PHOEBE_ast_list *args;
		} node;
	} val;
} PHOEBE_ast;
